3、this指向及Math的两个方法[.sqrt(), .pow()]（平方根，次方）

            var p = [
                {x: 0, y: 0},
                {x: 1, y: 1}
            ];
            p.dist = function(){
                var p1 = this[0];
                console.log(p1)
                // this指向调用者，p调用dist方法，所以p是调用者
            
                var p2 = this[1];
                var a = p2.x - p1.x;
                var b = p2.y - p1.y;
            
                console.log(Math.pow(2, 3)) // 补充：.pow(底数, 幕数)求次方
                return Math.sqrt(a*a + b*b);    // sqrt是求平方根
            }
